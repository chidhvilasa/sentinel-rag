### **Step 4: Integrate with Main Detector**

Update your main detection pipeline (e.g., `src/sentinel/detector.py`):

```python
from sentinel.multilang_detector import MultiLangDetector
from sentinel.adversarial_tester import AdversarialTester
from sentinel.context_neutralizer import ContextAwareNeutralizer
from sentinel.explainer import DetectionExplainer
from sentinel.zeroshot_detector import ZeroShotDetector

class SentinelV5:
    def __init__(self):
        # V4 detectors
        self.text_detector = ...
        self.image_detector = ...
        
        # V5 additions
        self.multilang_detector = MultiLangDetector()
        self.zeroshot_detector = ZeroShotDetector()
        self.neutralizer = ContextAwareNeutralizer()
        self.explainer = DetectionExplainer()
        
    def detect(self, text):
        results = {}
        
        # Run all detectors
        results['text'] = self.text_detector.detect(text)
        results['multilang'] = self.multilang_detector.detect(text)
        results['zeroshot'] = self.zeroshot_detector.detect(text)
        
        # Aggregate results
        is_attack = any(r.get('is_attack') for r in results.values())
        
        # Generate explanation
        explanation = self.explainer.explain(results, text)
        
        return {
            'is_attack': is_attack,
            'results': results,
            'explanation': explanation
        }
```
